Table Institucion {
  id int [pk, increment]
  email varchar [unique]
  passwordHash varchar
  nombreInstitucion varchar
  nit varchar [unique]
  direccion text
  telefono varchar
  pais varchar
  ciudad varchar
  suscripcion int
  codigoOTP varchar
  activo EstadoActivo
  fechaRegistro datetime
}

Table Plan {
  id int [pk, increment]
  nombre varchar
  codigo varchar
  price decimal
  currency varchar
  interval IntervaloPlan
  maxDays int
  createdAt datetime
  estado EstadoGeneral
}

Table Usuario {
  id int [pk, increment]
  nombreCompleto varchar
  email varchar [unique]
  passwordHash varchar
  telefono varchar
  ciudad varchar
  rol RolUsuario
  fechaNacimiento datetime
  nivelEducativo varchar
  institucionId int
  codigoOTP varchar
  activo EstadoActivo
  fechaRegistro datetime
}

Table InstitucionPlan {
  id int [pk, increment]
  institucionId int
  planId int
  periodStart datetime
  periodEnd datetime
  estado EstadoSuscripcion
}

Table Pago {
  id int [pk, increment]
  institucionId int
  planId int
  institucionPlanId int
  monto decimal
  currency varchar
  metodoPago MetodoPago
  tipoPago TipoPago
  estado EstadoPago
  referenciaPago varchar [unique]
  banco varchar
  reciboFile varchar
  comprobanteUrl text
  fechaPago datetime
  fechaVerificacion datetime
}

Table Session {
  id int [pk, increment]
  ip varchar
  usuarioId int
  fecha datetime
  navegador varchar
  status EstadoSesion
}

Table Curso {
  id int [pk, increment]
  institucionId int
  profesorId int
  img varchar
  nombreCurso varchar
  categoria CategoriaCurso
  modulo ModuloCurso
  idioma varchar
  descripcion text
  fechaInicio datetime
  fechaFin datetime
  estado EstadoCurso
  fechaCreacion datetime
}

Table Clase {
  id int [pk, increment]
  cursoId int
  estudianteId int
  img varchar
  titulo varchar
  descripcion text
  fechaClase datetime
  linkMeet varchar
  duracionMinutos int
  claseActual ClaseActual
  estadoClase EstadoClase
}

Table MaterialEstudio {
  id int [pk, increment]
  titulo varchar
  detalle text
  linkMaterial varchar
  tipo TipoMaterial
  claseActual ClaseActual
  claseMaterialId int
  estadoMaterial EstadoMaterial
}

Table Evaluacion {
  id int [pk, increment]
  cursoId int
  titulo varchar
  descripcion text
  codigoEvaluacion varchar
  fechaCreacion datetime
  fechaLimite datetime
  porcentajeCurso decimal
}

Table Pregunta {
  id int [pk, increment]
  evaluacionId int
  pregunta text
  respuestaA varchar
  respuestaB varchar
  respuestaC varchar
  respuestaD varchar
  respuestaCorrecta varchar
  puntaje float
  status EstadoGeneral
}

Table Resultado {
  id int [pk, increment]
  evaluacionId int
  estudianteId int
  respuestasJson text
  puntajeObtenido decimal
  fechaRealizacion datetime
}

Table Matricula {
  id int [pk, increment]
  cursoId int
  estudianteId int
  fechaMatricula datetime
  estado EstadoMatricula
}

Table Reporte {
  id int [pk, increment]
  cursoId int
  estudianteId int
  fechaGeneracion datetime
  promedioFinal decimal
  aprobado boolean
  codigoCertificado varchar [unique]
  pdfUrl text
}
